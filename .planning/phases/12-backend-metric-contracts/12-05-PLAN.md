    # Phase 12 / METRICS-07: derived trade metrics
    win_rate = result.get("win_rate", 0.0) or 0.0
    avg_win = result.get("avg_winning_trade", 0.0) or 0.0
    avg_loss = result.get("avg_losing_trade", 0.0) or 0.0

    # Expectancy = expected $ per trade given hit rate
    if avg_win or avg_loss:
        result["expectancy"] = win_rate * avg_win - (1 - win_rate) * abs(avg_loss)
    else:
        result["expectancy"] = None

    # Risk:Reward Ratio
    if avg_loss != 0:
        result["risk_reward_ratio"] = avg_win / abs(avg_loss)
    else:
        result["risk_reward_ratio"] = None

    # METRICS-08: SQN (Van Tharp)
    # R-multiples = trade_return / risk_unit, where risk_unit = abs(avg_loss)
    risk_unit = abs(avg_loss) if avg_loss else 0.0
    if risk_unit > 0:
        r_multiples = [
            (t.get("realized_pnl_usd", 0.0) or 0.0) / risk_unit
            for t in fills
            if t.get("realized_pnl_usd") is not None
        ]
        if len(r_multiples) >= 2:
            mean_r = sum(r_multiples) / len(r_multiples)
            var_r = sum((r - mean_r) ** 2 for r in r_multiples) / (len(r_multiples) - 1)
            std_r = math.sqrt(var_r) if var_r > 0 else 0.0
            if std_r > 0:
                result["sqn"] = (mean_r / std_r) * math.sqrt(min(len(r_multiples), 100))
            else:
                result["sqn"] = None
        else:
            result["sqn"] = None
    else:
        result["sqn"] = None

    # Profit Factor segmented by side
    long_pnls = [t.get("realized_pnl_usd", 0.0) or 0.0 for t in fills if t.get("side") == "long"]
    short_pnls = [t.get("realized_pnl_usd", 0.0) or 0.0 for t in fills if t.get("side") == "short"]

    def _profit_factor(pnls: list[float]) -> float | None:
        gross_profit = sum(p for p in pnls if p > 0)
        gross_loss = abs(sum(p for p in pnls if p < 0))
        if gross_loss == 0:
            return None  # Avoid +Infinity; downstream renders as "—"
        return gross_profit / gross_loss

    result["profit_factor_long"] = _profit_factor(long_pnls)
    result["profit_factor_short"] = _profit_factor(short_pnls)
    ```

       def _compute_volume_aggregator(fills: list[dict]) -> dict[str, float]:
           """METRICS-09: aggregate volume metrics over fills (raw_fills WHERE is_fill=true).

           Returns gross_volume_usd, mean_trade_size_usd, daily_turnover_usd, monthly_turnover_usd.
           """
           if not fills:
               return {
                   "gross_volume_usd": 0.0,
                   "mean_trade_size_usd": 0.0,
                   "daily_turnover_usd": 0.0,
                   "monthly_turnover_usd": 0.0,
               }
           notionals = [abs(f.get("notional_usd", 0.0) or 0.0) for f in fills]
           gross_volume = sum(notionals)
           mean_size = gross_volume / len(notionals) if notionals else 0.0

           # Daily / monthly turnover — group by day/month, then mean
           from collections import defaultdict
           daily: dict[str, float] = defaultdict(float)
           monthly: dict[str, float] = defaultdict(float)
           for f in fills:
               ts = f.get("filled_at") or f.get("created_at") or ""
               if not ts or len(ts) < 10:
                   continue
               day = ts[:10]
               month = ts[:7]
               daily[day] += abs(f.get("notional_usd", 0.0) or 0.0)
               monthly[month] += abs(f.get("notional_usd", 0.0) or 0.0)
           daily_avg = sum(daily.values()) / len(daily) if daily else 0.0
           monthly_avg = sum(monthly.values()) / len(monthly) if monthly else 0.0

           return {
               "gross_volume_usd": gross_volume,
               "mean_trade_size_usd": mean_size,
               "daily_turnover_usd": daily_avg,
               "monthly_turnover_usd": monthly_avg,
           }
       ```

    2. **`_compute_trade_mix(fills, has_maker_taker)`**:
       ```python
       def _compute_trade_mix(fills: list[dict], has_maker_taker: bool) -> dict[str, dict[str, float]]:
           """METRICS-10: Trade Mix breakdown by side × maker/taker.

           D-14 / D-15: bucket count branches off the is_maker audit outcome.
           - has_maker_taker=True  → 4 buckets (long_maker, long_taker, short_maker, short_taker)
           - has_maker_taker=False → 2 buckets fallback (long, short)
           """
           def _empty_bucket() -> dict[str, float]:
               return {"count": 0, "total_notional": 0.0, "avg_holding_period_hours": 0.0}

           if has_maker_taker:
               buckets: dict[str, dict[str, float]] = {
                   "long_maker": _empty_bucket(),
                   "long_taker": _empty_bucket(),
                   "short_maker": _empty_bucket(),
                   "short_taker": _empty_bucket(),
               }
           else:
               buckets = {
                   "long": _empty_bucket(),
                   "short": _empty_bucket(),
               }

           # Accumulate
           holding_sums: dict[str, float] = {k: 0.0 for k in buckets}
           for f in fills:
               side = f.get("side")
               if side not in ("long", "short"):
                   continue
               notional = abs(f.get("notional_usd", 0.0) or 0.0)
               holding = float(f.get("holding_period_hours") or 0.0)

               if has_maker_taker:
                   is_maker = f.get("is_maker")
                   if is_maker is None:
                       continue  # Skip fills without is_maker flag
                   maker_key = "maker" if is_maker else "taker"
                   bucket_key = f"{side}_{maker_key}"
               else:
                   bucket_key = side

               buckets[bucket_key]["count"] += 1
               buckets[bucket_key]["total_notional"] += notional
               holding_sums[bucket_key] += holding

           # Finalize avg holding period
           for k, b in buckets.items():
               if b["count"] > 0:
                   b["avg_holding_period_hours"] = holding_sums[k] / b["count"]

           return buckets
       ```

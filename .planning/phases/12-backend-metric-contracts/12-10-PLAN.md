    **Step 1:** Create `analytics-service/scripts/analyze_metrics_size.sql`:

    ```sql
    -- Phase 12 / D-07: probe metrics_json row size at p99.9 across all published strategies.
    -- If p99.9 >= 800kB, kill-switch (phase12_kill_switch.py) cuts over heavy keys to sibling table.
    --
    -- Usage:
    --   psql "$DATABASE_URL" -f analyze_metrics_size.sql
    SELECT
        percentile_cont(0.50) WITHIN GROUP (ORDER BY pg_column_size(metrics_json)) AS p50_bytes,
        percentile_cont(0.95) WITHIN GROUP (ORDER BY pg_column_size(metrics_json)) AS p95_bytes,
        percentile_cont(0.99) WITHIN GROUP (ORDER BY pg_column_size(metrics_json)) AS p99_bytes,
        percentile_cont(0.999) WITHIN GROUP (ORDER BY pg_column_size(metrics_json)) AS p999_bytes,
        max(pg_column_size(metrics_json)) AS max_bytes,
        count(*) AS strategy_count
    FROM strategy_analytics
    WHERE metrics_json IS NOT NULL;
    ```

    **Step 2:** Create `analytics-service/scripts/phase12_kill_switch.py`:

    ```python
    """Phase 12 / D-07: kill-switch — cut over heavy keys from metrics_json to sibling table
    when p99.9 ≥ 800kB.

    Usage:
        python -m scripts.phase12_kill_switch
        SKIP_KILL_SWITCH=1 python -m scripts.phase12_kill_switch  # honors override

    Idempotent: re-running after a successful cutover is a no-op (heavy keys already missing
    from metrics_json, INSERT...ON CONFLICT into sibling table is upsert).
    """
    from __future__ import annotations
    import json
    import os
    import sys
    from pathlib import Path

    from services.supabase_client import get_supabase
    from services.db_execute import db_execute  # adjust to actual module path

    THRESHOLD_BYTES = 800_000  # 800kB per Phase 12 SC#3a
    HEAVY_KINDS = [
        "daily_returns_grid",
        "rolling_sortino_3m", "rolling_sortino_6m", "rolling_sortino_12m",
        "rolling_volatility_3m", "rolling_volatility_6m", "rolling_volatility_12m",
        "rolling_alpha", "rolling_beta",
        "exposure_series", "turnover_series", "log_returns_series",
    ]


    # M-03 from 12-REVIEWS.md: NEVER measure size via Python json round-trip — that
    # measures pre-TOAST raw JSON bytes which can differ 30-50% from on-disk pg_column_size.
    # Instead: receive p999 as a CLI argument from `phase12_deploy.py` (which runs the SQL probe
    # authoritatively) OR re-run the SQL via raw query / supabase RPC.

    async def measure_p999_via_sql() -> tuple[float, int]:
        """Run the analyze_metrics_size.sql probe via service-role raw SQL — DB-side measurement only.

        Returns (p999_bytes, strategy_count). Uses pg_column_size which accounts for TOAST
        compression — the only authoritative size measurement.
        """
        # supabase-py does not expose raw-SQL-text execution by default; the canonical path is to
        # ship `analyze_metrics_size.sql` as a SECURITY DEFINER RPC in a Phase 12 migration footer
        # (e.g. `analyze_metrics_size_p999()` returning a single row).
        # If that RPC is not available at execution time, fall back to invoking psql via subprocess
        # against the service-role DATABASE_URL (env var must be set on the deploy host).
        import subprocess
        sql = (Path(__file__).parent / "analyze_metrics_size.sql").read_text()
        db_url = os.getenv("DATABASE_URL") or os.getenv("SUPABASE_DB_URL")
        if not db_url:
            raise RuntimeError("phase12_kill_switch: DATABASE_URL not set; cannot run pg_column_size SQL probe")
        result = subprocess.run(
            ["psql", db_url, "-tAF,", "-c", sql],
            capture_output=True, text=True, check=False,
        )
        if result.returncode != 0:
            raise RuntimeError(f"phase12_kill_switch: SQL probe failed: {result.stderr}")
        # Parse: p50_bytes,p95_bytes,p99_bytes,p999_bytes,max_bytes,strategy_count
        line = result.stdout.strip().splitlines()[-1] if result.stdout.strip() else ""
        parts = line.split(",")
        if len(parts) < 6:
            raise RuntimeError(f"phase12_kill_switch: unexpected SQL output: {line}")
        p999 = float(parts[3]) if parts[3] else 0.0
        n = int(parts[5]) if parts[5] else 0
        return (p999, n)


    async def measure_p999(cli_p999: float | None = None, cli_count: int | None = None) -> tuple[float, int]:
        """Returns (p999_bytes, strategy_count). Prefers caller-provided values from
        `phase12_deploy.py` SQL probe (M-03); falls back to direct SQL run.
        """
        if cli_p999 is not None and cli_count is not None:
            return (cli_p999, cli_count)
        return await measure_p999_via_sql()


    async def cutover_strategy(strategy_id: str) -> int:
        """Move heavy keys from metrics_json to sibling table for one strategy. Per-strategy txn."""
        supabase = get_supabase()
        # Read current metrics_json
        result = await db_execute(
            lambda: supabase.table("strategy_analytics")
            .select("metrics_json")
            .eq("strategy_id", strategy_id)
            .single()
            .execute()
        )
        if not result.data or not result.data.get("metrics_json"):
            return 0
        m = result.data["metrics_json"]
        moved = 0
        for kind in HEAVY_KINDS:
            if kind in m:
                payload = m[kind]
                # Upsert to sibling
                await db_execute(
                    lambda k=kind, p=payload: supabase.table("strategy_analytics_series").upsert(
                        {"strategy_id": strategy_id, "kind": k, "payload": p},
                        on_conflict="strategy_id,kind",
                    ).execute()
                )
                # Remove from metrics_json
                del m[kind]
                moved += 1
        if moved > 0:
            await db_execute(
                lambda: supabase.table("strategy_analytics")
                .update({"metrics_json": m})
                .eq("strategy_id", strategy_id)
                .execute()
            )
        return moved


    async def main(cli_p999: float | None = None, cli_count: int | None = None) -> int:
        if os.getenv("SKIP_KILL_SWITCH") == "1":
            print("phase12_kill_switch: SKIP_KILL_SWITCH=1 — bypassing.")
            return 0

        # M-03: prefer caller-provided p999 from SQL probe; fall back to direct SQL re-run.
        # NEVER use Python json size approximation — bypasses TOAST compression, mis-measures by 30-50%.
        p999, n = await measure_p999(cli_p999=cli_p999, cli_count=cli_count)
        print(f"phase12_kill_switch: probe — p99.9 = {p999:.0f} bytes across {n} strategies (threshold {THRESHOLD_BYTES}) [M-03: pg_column_size, DB-side only]")

        if p999 < THRESHOLD_BYTES:
            print(f"phase12_kill_switch: p99.9 < threshold — no cutover needed.")
            return 0

        print(f"phase12_kill_switch: TRIGGERED — cutting over heavy keys for all strategies")
        # Get all strategy_ids
        supabase = get_supabase()
        rows = await db_execute(
            lambda: supabase.table("strategy_analytics").select("strategy_id").execute()
        )
        total_moved = 0
        for r in rows.data:
            moved = await cutover_strategy(r["strategy_id"])
            total_moved += moved
            print(f"  strategy {r['strategy_id']}: moved {moved} keys")

        print(f"phase12_kill_switch: COMPLETE — {total_moved} keys moved across {len(rows.data)} strategies")
        # STATE.md log entry — append a TODOS line
        log_path = Path(".planning/phases/12-backend-metric-contracts/TODOS.md")
        if log_path.exists():
            log_path.write_text(
                log_path.read_text() + f"\n## Kill-switch triggered (D-07)\n"
                f"- p99.9 = {p999:.0f} bytes; moved {total_moved} keys; {len(rows.data)} strategies updated.\n"
            )
        return 0


    if __name__ == "__main__":
        import asyncio
        import argparse
        # M-03: accept p999 + strategy_count from deploy orchestrator
        parser = argparse.ArgumentParser(description="Phase 12 kill-switch — DB-side pg_column_size measurement only (M-03)")
        parser.add_argument("--p999", type=float, default=None, help="pg_column_size p99.9 in bytes (from phase12_deploy.py SQL probe)")
        parser.add_argument("--count", type=int, default=None, help="strategy_count from SQL probe")
        args = parser.parse_args()
        sys.exit(asyncio.run(main(cli_p999=args.p999, cli_count=args.count)))
    ```

    **Step 3:** Create `analytics-service/scripts/phase12_backfill_enqueue.py`:

    ```python
    """Phase 12 / D-08: eager re-enqueue all published strategies as priority=low.

    Throttle (Plan 07) caps backfill at 5/min when sync_trades is queued, so this
    blast cannot starve live ingest.

    Usage:
        python -m scripts.phase12_backfill_enqueue

    Idempotent: re-running enqueues again, but the deduplication is at the worker level
    (`compute_jobs` unique constraint on (strategy_id, kind, payload) — verify before running).
    """
    from __future__ import annotations
    import asyncio
    import sys
    from datetime import datetime, timezone

    from services.supabase_client import get_supabase
    from services.db_execute import db_execute  # adjust path


    async def main() -> int:
        supabase = get_supabase()

        # M-02 from 12-REVIEWS.md: compute_jobs (per migration 032) has NO unique constraint on
        # (strategy_id, kind) — re-running enqueue without pre-check would insert duplicate
        # `compute_analytics` jobs. Pre-check first; abort with notice if pending jobs exist.
        existing = await db_execute(
            lambda: supabase.table("compute_jobs")
            .select("id", count="exact")
            .eq("kind", "compute_analytics")
            .eq("status", "pending")
            .execute()
        )
        if (existing.count or 0) > 0:
            print(
                f"[backfill] {existing.count} existing pending compute_analytics jobs found — "
                f"skipping to avoid duplicates. Re-run after worker drains."
            )
            return 0

        # Fetch all published strategy IDs
        rows = await db_execute(
            lambda: supabase.table("strategies")
            .select("id")
            .eq("status", "published")
            .execute()
        )
        n = len(rows.data) if rows.data else 0
        print(f"phase12_backfill_enqueue: enqueueing {n} published strategies as priority=low")

        for r in rows.data or []:
            await db_execute(
                lambda strategy_id=r["id"]: supabase.table("compute_jobs").insert(
                    {
                        "kind": "compute_analytics",
                        "status": "pending",
                        "priority": "low",
                        "payload": {"strategy_id": strategy_id, "phase": "12-backfill"},
                        "next_attempt_at": datetime.now(timezone.utc).isoformat(),
                    }
                ).execute()
            )

        print(f"phase12_backfill_enqueue: enqueued {n} jobs. Throttle (5/min when sync_trades queued) will pace.")
        return 0


    if __name__ == "__main__":
        sys.exit(asyncio.run(main()))
    ```

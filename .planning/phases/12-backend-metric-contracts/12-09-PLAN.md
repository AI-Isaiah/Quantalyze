    ```python
    FIXTURES_DIR = Path(__file__).parent
    SEED = 42
    N_DAYS = 252


    def _build_input() -> dict:
        np.random.seed(SEED)
        dates = pd.bdate_range("2025-01-01", periods=N_DAYS)

        # Calibrated random walk: ~0.4 Sharpe, ~5% annualized vol, ~10% max DD
        base = np.random.normal(0.0008, 0.013, N_DAYS)
        # Inject controlled drawdown around day 120
        base[120:140] = np.random.normal(-0.005, 0.015, 20)
        returns = pd.Series(base, index=dates, name="returns")

        # BTC benchmark
        bench = np.random.normal(0.0003, 0.025, N_DAYS)
        benchmark = pd.Series(bench, index=dates, name="benchmark")

        # Synthetic fills (250 trades) and positions (50 closed)
        fills: list[dict] = []
        for i in range(250):
            day_idx = (i * 5) % N_DAYS
            side = "long" if i % 2 == 0 else "short"
            is_maker = (i % 3 == 0)  # 33% maker, 67% taker
            fills.append({
                "filled_at": dates[day_idx].strftime("%Y-%m-%d"),
                "side": side,
                "is_maker": is_maker,
                "is_fill": True,
                "notional_usd": 1000.0 * (1 + (i % 10) / 10.0),
                "realized_pnl_usd": float(np.random.normal(2.0, 50.0)),
                "holding_period_hours": 4.0 + (i % 20),
            })

        return {
            "returns": returns,
            "benchmark": benchmark,
            "fills": fills,
        }


    def _serialize_series(s: pd.Series) -> list[dict]:
        return [
            {"date": d.strftime("%Y-%m-%d"), "value": float(v)}
            for d, v in s.items()
        ]


    def main() -> None:
        inp = _build_input()

        # Parquet for Python side
        pd.DataFrame({
            "returns": inp["returns"],
            "benchmark": inp["benchmark"],

        input_json = {

            "fills": inp["fills"],

            "metrics_json": metrics_json,
            "sibling": sibling,

            json.dumps(expected, indent=2, sort_keys=True, default=_json_default)
        )

        print(f"Fixtures regenerated. has_maker_taker={has_maker_taker}")
        print(f"  metrics_json keys: {len(metrics_json)}")
        print(f"  sibling kinds: {len(sibling)}")


    def _json_default(obj):
        if isinstance(obj, (np.floating, np.integer)):
            return float(obj)
        if isinstance(obj, np.ndarray):
            return obj.tolist()
        if isinstance(obj, pd.Timestamp):
            return obj.strftime("%Y-%m-%d")
        raise TypeError(f"Not JSON-serializable: {type(obj)}")


    if __name__ == "__main__":
        main()
    ```

    ```python
    """METRICS-13: Python-side parity test.

    Reads input parquet, runs compute_all_metrics(), compares against expected JSON.
    D-11 hybrid tolerance:
    - Scalar keys: byte-identical after rounding to 12 significant digits
    - Series values: 1e-9 relative epsilon (NaN==NaN, +0==-0)
    - Missing or extra keys: FAIL (D-12 fail-loud on contract drift)

    Run: cd analytics-service && pytest tests/test_metrics_parity.py -x
    """
    from __future__ import annotations
    import json
    import math
    import os
    from pathlib import Path
    from typing import Any

    import pytest

    from services.metrics import compute_all_metrics
    from services.analytics_runner import _compute_volume_metrics, _compute_volume_aggregator, _compute_trade_mix

    FIXTURES_DIR = Path(__file__).parent / "fixtures"


    def _round_sig(x: float, sig: int = 12) -> float:
        """Round to N significant digits (D-11 scalar comparator)."""
        if x == 0 or math.isnan(x) or math.isinf(x):
            return x
        return round(x, -int(math.floor(math.log10(abs(x)))) + (sig - 1))


    def _assert_scalar_equal(actual: Any, expected: Any, key_path: str) -> None:
        """D-11 scalar: byte-identical after 12-sig-digit rounding."""
        if expected is None:
            assert actual is None, f"{key_path}: expected None, got {actual}"
            return
        if isinstance(expected, bool):
            assert actual == expected, f"{key_path}: expected {expected}, got {actual}"
            return
        if isinstance(expected, (int, float)):
            if math.isnan(expected):
                assert math.isnan(actual), f"{key_path}: expected NaN, got {actual}"
                return
            assert isinstance(actual, (int, float)), f"{key_path}: expected number, got {type(actual)}"
            assert _round_sig(float(actual)) == _round_sig(float(expected)), (
                f"{key_path}: scalar mismatch (12 sig digits): expected {expected}, got {actual}"
            )
            return
        # Strings, dicts handled at higher level
        assert actual == expected, f"{key_path}: deep mismatch — expected {expected!r}, got {actual!r}"


    def _assert_series_equal(actual: list, expected: list, key_path: str, eps: float = 1e-9) -> None:
        """D-11 series: 1e-9 relative epsilon, NaN==NaN, +0==-0."""
        assert len(actual) == len(expected), (
            f"{key_path}: series length {len(actual)} != expected {len(expected)}"
        )
        for i, (a, e) in enumerate(zip(actual, expected)):
            if isinstance(e, dict) and "value" in e:
                assert a.get("date") == e["date"], (
                    f"{key_path}[{i}]: date mismatch {a.get('date')} != {e['date']}"
                )
                if e["value"] is None:
                    assert a["value"] is None, f"{key_path}[{i}]: expected None, got {a['value']}"
                elif isinstance(e["value"], float) and math.isnan(e["value"]):
                    assert math.isnan(a["value"]), f"{key_path}[{i}]: expected NaN"
                else:
                    rel = abs(a["value"] - e["value"]) / max(abs(e["value"]), 1.0)
                    assert rel < eps, (
                        f"{key_path}[{i}]: series rel-eps mismatch: expected {e['value']}, got {a['value']} (rel={rel})"
                    )
            else:
                _assert_scalar_equal(a, e, f"{key_path}[{i}]")


    def assertMetricParity(actual: dict, expected: dict, prefix: str = "") -> None:
        """D-11 + D-12 enforcement.

        - Scalar keys: byte-identical after rounding to 12 sig digits
        - Series values: 1e-9 relative epsilon
        - NaN==NaN, +0==-0
        - Missing key in actual that's present in expected: fail
        - Extra key in actual not in expected: fail (forces fixture regen on metric add)
        """
        actual_keys = set(actual.keys())
        expected_keys = set(expected.keys())
        missing = expected_keys - actual_keys
        extra = actual_keys - expected_keys
        assert not missing, f"{prefix}: missing keys in actual: {sorted(missing)}"
        assert not extra, f"{prefix}: extra keys in actual: {sorted(extra)} (regen golden_252d_expected.json?)"

        for key in expected_keys:
            kp = f"{prefix}.{key}" if prefix else key
            a = actual[key]
            e = expected[key]
            if isinstance(e, list):
                _assert_series_equal(a, e, kp)
            elif isinstance(e, dict):
                assertMetricParity(a, e, prefix=kp)
            else:
                _assert_scalar_equal(a, e, kp)


    def test_metrics_parity_full(golden_252d_input, golden_252d_expected):
        """METRICS-13: full parity assertion against committed fixture."""
        # Run the actual metrics path
        result = compute_all_metrics(golden_252d_input["returns"], golden_252d_input["benchmark"])
        # Reconstruct the same output structure as regen_golden.py
        # Read fills from input.json companion for trade-side metrics
        import json
        input_json = json.loads((FIXTURES_DIR / "golden_252d_input.json").read_text())
        fills = input_json["fills"]

        metrics_json = dict(result.metrics_json)
        metrics_json["trade_metrics"] = _compute_volume_metrics(fills)
        metrics_json["volume_metrics"] = _compute_volume_aggregator(fills)
        has_maker_taker = os.getenv("TRADE_MIX_HAS_MAKER_TAKER") == "true"
        metrics_json["trade_metrics"]["trade_mix"] = _compute_trade_mix(fills, has_maker_taker=has_maker_taker)

        actual = {
            "metrics_json": metrics_json,
            "sibling": dict(result.sibling_kinds),
        }
        assertMetricParity(actual, golden_252d_expected)
    ```

    ```typescript
     * Asserts the trade_mix bucket count matches the D-15 audit outcome.

      metrics_json: Record<string, unknown>;
    }): void {
      const tradeMix = (expected.metrics_json["trade_metrics"] as Record<string, unknown> | undefined)?.["trade_mix"];
      if (!tradeMix || typeof tradeMix !== "object") {

      }
      const keys = Object.keys(tradeMix as Record<string, unknown>);
      const hasMakerTaker = process.env.TRADE_MIX_HAS_MAKER_TAKER === "true";
      if (hasMakerTaker) {
        const expected4 = ["long_maker", "long_taker", "short_maker", "short_taker"];
        for (const k of expected4) {

              `4-bucket Trade Mix missing key: ${k}. Got ${keys.join(", ")}. ` +
              `Audit reported is_maker available; expected 4 buckets.`,
            );
          }
        }
      } else {
        const expected2 = ["long", "short"];

            throw new Error(
              `2-bucket fallback missing key: ${k}. Got ${keys.join(", ")}. ` +
              `Audit reported is_maker unavailable; expected 2 buckets.`,
            );
          }
        }
      }
    }
    ```

```tsx
"use client";

/**
 * Phase 15 / CSV-02: validation envelope renderer for the CSV branch
 * of the strategy onboarding wizard.
 *
 * Visual contract locked by UI-SPEC §7.3 — mirrors the ConnectKeyStep
 * error envelope (rounded-md border border-negative/30 bg-negative/5
 * px-4 py-3) with a collapsible <details> per rule category. Native
 * HTML <details> — no JS state machine, no animation.
 *
 * Phase 16 / OBSERV-06 will wire correlation_id; the slot is rendered
 * here as "—" placeholder so the DOM shape stays stable.
 */

interface CsvValidationEnvelopeProps {
  envelope: {
    code: string;
    human_message: string;
    debug_context: {
      pandera_errors?: { rule: string; row: number; message: string }[];
    };
    correlation_id: string | null;
  };
}

// TODO(phase-17): hoist into wizardErrors.ts per DESIGN-05.
// UI-SPEC §8.7 rule-name human mapping (locked).
const RULE_LABELS: Record<string, string> = {
  monotonic_dates: "Dates must be strictly increasing",
  nav_non_zero: "NAV cannot be zero",
  daily_return_lower_bound: "Daily return cannot be ≤ -100%",
  daily_sharpe_sentinel: "Daily Sharpe > 10 looks unrealistic",
  trading_window: "Date falls outside your trading window",
  currency_usd_or_blank: "Currency must be USD or left blank",
  qty_price_positive: "Quantity and price must be positive",
};

export function CsvValidationEnvelope({ envelope }: CsvValidationEnvelopeProps) {
  const errors = envelope.debug_context?.pandera_errors ?? [];
  const byRule = errors.reduce<Record<string, typeof errors>>((acc, e) => {
    (acc[e.rule] ??= []).push(e);
    return acc;
  }, {});
  const ruleCount = Object.keys(byRule).length;

  return (
    <div
      role="alert"
      className="rounded-md border border-negative/30 bg-negative/5 px-4 py-3"
      data-testid="wizard-csv-error"
      data-error-code={envelope.code}
    >
      <p className="text-sm font-semibold text-negative">
        {errors.length > 0
          ? `${errors.length} ${errors.length === 1 ? "row" : "rows"} failed validation`
          : envelope.human_message}
      </p>
      <p className="mt-1 text-xs text-text-secondary">
        {ruleCount > 1
          ? `Across ${ruleCount} rule categories: ${Object.keys(byRule)
              .map((r) => RULE_LABELS[r] ?? r)
              .join(", ")}.`
          : envelope.human_message}
      </p>
      {Object.entries(byRule).map(([rule, list]) => (
        <details key={rule} className="mt-2 text-xs">
          <summary className="cursor-pointer text-text-secondary">
            {RULE_LABELS[rule] ?? rule} ({list.length} rows)
          </summary>
          <ul className="mt-1 list-disc space-y-0.5 pl-5 text-text-muted">
            {list.map((e, i) => (
              <li key={i}>
                Row {e.row}: {e.message}
              </li>
            ))}
          </ul>
        </details>
      ))}
      {/* Phase 16 / OBSERV-06 carrier marker — DOM shape stable. */}
      <p className="mt-2 text-[11px] text-text-muted">
        correlation_id: {envelope.correlation_id ?? "—"}
      </p>
    </div>
  );
}
```

```tsx
"use client";

import { useCallback, useRef, useState } from "react";
import { Button } from "@/components/ui/Button";
import { trackForQuantsEventClient } from "@/lib/for-quants-analytics";
import { CsvValidationEnvelope } from "./CsvValidationEnvelope";

type CsvFmt = "daily_returns" | "daily_nav" | "trades";

// TODO(phase-17): hoist into wizardErrors.ts per DESIGN-05.
const FORMATS: { id: CsvFmt; label: string; caption: string }[] = [
  { id: "daily_returns", label: "Daily returns", caption: "One row per trading day. Columns: date, daily_return." },
  { id: "daily_nav",     label: "Daily NAV",     caption: "One row per trading day. Columns: date, nav." },
  { id: "trades",        label: "Trade list",    caption: "One row per fill. Columns: date, side, qty, price, symbol, currency." },
];

const MAX_BYTES = 10 * 1024 * 1024;

interface PreviewShape {
  row_count: number;
  date_range: [string, string];
  columns_detected: string[];
  first_rows: Record<string, unknown>[];
  last_rows: Record<string, unknown>[];
}

interface CsvUploadStepProps {
  wizardSessionId: string;
  onSuccess: (payload: { fmt: CsvFmt; preview: PreviewShape; validationPassed: boolean }) => void;
}

export function CsvUploadStep({ wizardSessionId, onSuccess }: CsvUploadStepProps) {
  const [fmt, setFmt] = useState<CsvFmt>("daily_returns");
  const [file, setFile] = useState<File | null>(null);
  const [submitting, setSubmitting] = useState(false);
  const [envelope, setEnvelope] = useState<{
    code: string;
    human_message: string;
    debug_context: { pandera_errors?: { rule: string; row: number; message: string }[] };
    correlation_id: string | null;
  } | null>(null);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleFileSelect = useCallback((selected: File | null) => {
    if (!selected) return;
    if (!selected.name.toLowerCase().endsWith(".csv")) {
      // TODO(phase-17): hoist into wizardErrors.ts.
      setEnvelope({
        code: "CSV_INVALID_EXTENSION",
        human_message: "Only .csv files are accepted. Convert your file and try again.",
        debug_context: {},
        correlation_id: null,
      });
      return;
    }
    if (selected.size > MAX_BYTES) {
      setEnvelope({
        code: "CSV_FILE_TOO_LARGE",
        human_message: `Maximum file size is 10 MB. Your file is ${(selected.size / 1024 / 1024).toFixed(1)} MB. Trim it or split it before retrying.`,
        debug_context: {},
        correlation_id: null,
      });
      return;
    }
    setEnvelope(null);
    setFile(selected);
  }, []);

  const handleSubmit = useCallback(async (e: React.FormEvent) => {
    e.preventDefault();
    if (submitting || !file) return;
    setSubmitting(true);
    setEnvelope(null);

    try {
      const formData = new FormData();
      formData.append("file", file);
      formData.append("fmt", fmt);
      formData.append("wizard_session_id", wizardSessionId);

      const res = await fetch("/api/strategies/csv-validate", {
        method: "POST",
        body: formData,
      });
      const data = await res.json().catch(() => ({}));

      if (!res.ok) {
        // The Next.js proxy returns the v0 envelope on error; surface verbatim.
        setEnvelope({
          code: data.code ?? "CSV_UPSTREAM_FAIL",
          human_message: data.human_message ?? "CSV validation failed.",
          debug_context: data.debug_context ?? {},
          correlation_id: data.correlation_id ?? null,
        });
        trackForQuantsEventClient("wizard_error", {
          wizard_session_id: wizardSessionId,
          step: "csv_upload",
          code: data.code ?? "UNKNOWN",
        });
        setSubmitting(false);
        return;
      }

      // Server returns 200 even for soft-fail (envelope ok=false).
      if (data.ok === false) {
        setEnvelope({
          code: "CSV_VALIDATION_FAILED",
          human_message: `${(data.errors ?? []).length} rows failed validation`,
          debug_context: { pandera_errors: data.errors ?? [] },
          correlation_id: data.correlation_id ?? null,
        });
        trackForQuantsEventClient("wizard_error", {
          wizard_session_id: wizardSessionId,
          step: "csv_upload",
          code: "CSV_VALIDATION_FAILED",
        });
        setSubmitting(false);
        return;
      }

      // Happy path — advance to Preview with the validation payload.
      onSuccess({ fmt, preview: data.preview as PreviewShape, validationPassed: true });
    } catch (err) {
      console.error("[wizard:CsvUploadStep] threw:", err);
      setEnvelope({
        code: "KEY_NETWORK_TIMEOUT",
        human_message: "The server did not respond within 30 seconds. Your file is preserved — click Retry to try again.",
        debug_context: {},
        correlation_id: null,
      });
      setSubmitting(false);
    }
  }, [submitting, file, fmt, wizardSessionId, onSuccess]);

  return (
    <section aria-labelledby="wizard-csv-upload-heading">
      <h2 id="wizard-csv-upload-heading" className="font-sans text-2xl font-semibold text-text-primary">
        Upload your track record
      </h2>
      <p className="mt-2 text-sm text-text-secondary">
        Pick a format and drop your CSV. We validate every row before creating your strategy. Max 10 MB.
      </p>

      <form onSubmit={handleSubmit} className="mt-6 space-y-6">
        <fieldset>
          <legend className="text-xs font-medium text-text-primary">Format</legend>
          <div className="mt-2 grid grid-cols-1 gap-2 md:grid-cols-3">
            {FORMATS.map((f) => {
              const active = f.id === fmt;
              return (
                <button
                  key={f.id}
                  type="button"
                  onClick={() => setFmt(f.id)}
                  className={`rounded-md border px-4 py-3 text-left transition-colors ${
                    active ? "border-accent bg-accent/5" : "border-border bg-white hover:border-accent/50"
                  }`}
                  aria-pressed={active}
                  data-testid={`wizard-csv-fmt-${f.id}`}
                >
                  <p className="text-sm font-semibold text-text-primary">{f.label}</p>
                  <p className="mt-1 text-[11px] text-text-muted">{f.caption}</p>
                </button>
              );
            })}
          </div>
        </fieldset>

        <div
          className="rounded-lg border-2 border-dashed border-border p-8 text-center transition-colors hover:border-accent/50"
          onClick={() => fileInputRef.current?.click()}
          onDragOver={(e) => e.preventDefault()}
          onDrop={(e) => {
            e.preventDefault();
            handleFileSelect(e.dataTransfer.files[0] ?? null);
          }}
          role="button"
          tabIndex={0}
          aria-label="Upload CSV file. Drop a file or press Enter to browse."
          onKeyDown={(e) => {
            if (e.key === "Enter" || e.key === " ") fileInputRef.current?.click();
          }}
          data-testid="wizard-csv-dropzone"
        >
          {file ? (
            <p className="text-sm text-text-primary">
              <span className="font-metric tabular-nums">{file.name}</span>
              {" · "}
              <span className="font-metric tabular-nums">{(file.size / 1024 / 1024).toFixed(1)} MB</span>
            </p>
          ) : (
            <>
              <p className="mb-1 text-sm text-text-muted">Drop a CSV file here, or click to browse</p>
              <p className="text-xs text-text-muted">Required columns shown above. Max 10 MB.</p>
            </>
          )}
          <input
            ref={fileInputRef}
            type="file"
            accept=".csv"
            className="hidden"
            onChange={(e) => handleFileSelect(e.target.files?.[0] ?? null)}
            data-testid="wizard-csv-file-input"
          />
        </div>

        {file && (
          <Button
            type="button"
            variant="ghost"
            onClick={() => {
              setFile(null);
              setEnvelope(null);
              if (fileInputRef.current) fileInputRef.current.value = "";
            }}
          >
            Choose a different file
          </Button>
        )}

        {envelope && <CsvValidationEnvelope envelope={envelope} />}

        <Button
          type="submit"
          disabled={submitting || !file}
          data-testid="wizard-csv-validate-submit"
        >
          {submitting ? "Validating…" : "Validate and continue"}
        </Button>
      </form>
    </section>
  );
}
```

```tsx
"use client";

import { useCallback, useState } from "react";
import { Button } from "@/components/ui/Button";
import { trackForQuantsEventClient } from "@/lib/for-quants-analytics";
import { CsvValidationEnvelope } from "./CsvValidationEnvelope";

type CsvFmt = "daily_returns" | "daily_nav" | "trades";

interface PreviewShape {
  row_count: number;
  date_range: [string, string];
  columns_detected: string[];
  first_rows: Record<string, unknown>[];
  last_rows: Record<string, unknown>[];

// TODO(phase-17): hoist into wizardErrors.ts per DESIGN-05.
const FMT_LABEL: Record<CsvFmt, string> = {
  daily_returns: "Daily returns",
  daily_nav: "Daily NAV",
  trades: "Trade list",
};

interface CsvSubmitStepProps {
  wizardSessionId: string;
  fmt: CsvFmt;
  preview: PreviewShape;
  onSubmitted: (strategyId: string) => void;
  onBack: () => void;
}

function SummaryRow({ label, value }: { label: string; value: string }) {
  return (
    <div className="grid gap-1 px-4 py-3 md:grid-cols-[180px_1fr] md:gap-6">
      <dt className="text-[11px] font-medium uppercase tracking-wider text-text-muted">{label}</dt>
      <dd className="text-xs text-text-secondary">{value}</dd>
    </div>
  );
}

export function CsvSubmitStep({
  wizardSessionId,
  fmt,
  preview,
  onSubmitted,
  onBack,
}: CsvSubmitStepProps) {
  const [submitting, setSubmitting] = useState(false);
  const [envelope, setEnvelope] = useState<{
    code: string;
    human_message: string;
    debug_context: { pandera_errors?: { rule: string; row: number; message: string }[] };
    correlation_id: string | null;
  } | null>(null);

  const handleSubmit = useCallback(async () => {
    if (submitting) return;
    setSubmitting(true);
    setEnvelope(null);
    try {
      const res = await fetch("/api/strategies/csv-finalize", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          wizard_session_id: wizardSessionId,
          fmt,
        }),
      });
      const data = await res.json().catch(() => ({}));
      if (!res.ok) {
        // TODO(phase-17): hoist into wizardErrors.ts.
        setEnvelope({
          code: data.code ?? "CSV_FINALIZE_FAIL",
          human_message: data.human_message ?? "Your file validated cleanly, but saving the strategy hit an error. Click Submit strategy again to retry — your data is unchanged.",
          debug_context: data.debug_context ?? {},
          correlation_id: data.correlation_id ?? null,
        });
        trackForQuantsEventClient("wizard_error", {
          wizard_session_id: wizardSessionId,
          step: "csv_submit",
          code: data.code ?? "CSV_FINALIZE_FAIL",
        });
        setSubmitting(false);
        return;
      }
      if (typeof data.strategy_id !== "string") {
        // Defensive: the route MUST return a strategy_id on success.
        // Surface a structured error rather than navigating to /strategies/undefined.
        setEnvelope({
          code: "CSV_FINALIZE_FAIL",
          human_message: "The server did not return a strategy id. Please retry.",
          debug_context: {},
          correlation_id: null,
        });
        setSubmitting(false);
        return;
      }
      trackForQuantsEventClient("wizard_step_complete_3", {
        wizard_session_id: wizardSessionId,
        strategy_id: data.strategy_id,
      });
      onSubmitted(data.strategy_id);
    } catch (err) {
      console.error("[wizard:CsvSubmitStep] threw:", err);
      setEnvelope({
        code: "KEY_NETWORK_TIMEOUT",
        human_message: "The server did not respond within 30 seconds. Your file is preserved — click Submit strategy again to retry.",
        debug_context: {},
        correlation_id: null,
      });
      setSubmitting(false);
    }
  }, [submitting, wizardSessionId, fmt, onSubmitted]);

  return (
    <section aria-labelledby="wizard-csv-submit-heading">
      <h2 id="wizard-csv-submit-heading" className="font-sans text-2xl font-semibold text-text-primary">
        Review and submit
      </h2>
      <p className="mt-2 text-sm text-text-secondary">
        The founder reviews CSV-uploaded strategies within 48 hours. You will receive an email when your listing is approved.
      </p>

      <div className="mt-6 rounded-lg border border-border bg-white">
        <dl className="divide-y divide-border">
          <SummaryRow label="Format" value={FMT_LABEL[fmt]} />
          <SummaryRow label="Rows detected" value={`${preview.row_count} rows`} />
          <SummaryRow
            label="Date range"
            value={`${preview.date_range[0]} → ${preview.date_range[1]}`}
          />
        </dl>
      </div>

      {envelope && <div className="mt-4"><CsvValidationEnvelope envelope={envelope} /></div>}

      <div className="mt-6 flex gap-3">
        <Button variant="secondary" type="button" onClick={onBack} data-testid="wizard-csv-submit-back">
          Back
        </Button>
        <Button
          type="button"
          onClick={handleSubmit}
          disabled={submitting}
          data-testid="wizard-csv-submit-cta"
        >
          {submitting ? "Submitting…" : "Submit strategy"}
        </Button>
      </div>
    </section>
  );
}
```

function makeMultipart(file: File | null, fmt: string | null) {
  const form = new FormData();
  if (file) form.append("file", file);
  if (fmt) form.append("fmt", fmt);
  form.append("wizard_session_id", "00000000-0000-0000-0000-000000000001");

    method: "POST",
    body: form,
  });

describe("/api/strategies/csv-validate", () => {
  beforeEach(() => {
    vi.clearAllMocks();
    (checkLimit as any).mockResolvedValue({ success: true, retryAfter: 0 });
  });

  it("happy path: validateCsv returns ok=true → 200 envelope verbatim", async () => {
    (validateCsv as any).mockResolvedValue({ ok: true, preview: { row_count: 5, date_range: ["2026-01-01","2026-01-05"], columns_detected: ["date","daily_return"], first_rows: [], last_rows: [] }, errors: [], correlation_id: null });

    expect(res.status).toBe(200);
    const json = await res.json();
    expect(json.ok).toBe(true);
    expect(json.preview.row_count).toBe(5);

  it("soft-fail: validateCsv returns ok=false → 200 with errors populated", async () => {

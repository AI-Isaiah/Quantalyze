function makeMultipart(file: File | null, fmt: string | null) {
  const form = new FormData();
  if (file) form.append("file", file);
  if (fmt) form.append("fmt", fmt);
  form.append("wizard_session_id", "00000000-0000-0000-0000-000000000001");
  return new NextRequest("http://localhost/api/strategies/csv-validate", {
    method: "POST",
    body: form,
  });
}

describe("/api/strategies/csv-validate", () => {
  beforeEach(() => {
    vi.clearAllMocks();
    (checkLimit as any).mockResolvedValue({ success: true, retryAfter: 0 });
  });

  it("happy path: validateCsv returns ok=true → 200 envelope verbatim", async () => {
    (validateCsv as any).mockResolvedValue({ ok: true, preview: { row_count: 5, date_range: ["2026-01-01","2026-01-05"], columns_detected: ["date","daily_return"], first_rows: [], last_rows: [] }, errors: [], correlation_id: null });
    const file = new File(["date,daily_return\n2026-01-01,0.01"], "x.csv", { type: "text/csv" });
    const req = makeMultipart(file, "daily_returns");
    const res = await POST(req);
    expect(res.status).toBe(200);
    const json = await res.json();
    expect(json.ok).toBe(true);
    expect(json.preview.row_count).toBe(5);
  });

  it("soft-fail: validateCsv returns ok=false → 200 with errors populated", async () => {
    (validateCsv as any).mockResolvedValue({ ok: false, preview: null, errors: [{ rule: "monotonic_dates", row: 2, message: "..." }], correlation_id: null });
    const file = new File(["bad"], "x.csv", { type: "text/csv" });
    const req = makeMultipart(file, "daily_returns");
    const res = await POST(req);
    expect(res.status).toBe(200);
    const json = await res.json();
    expect(json.ok).toBe(false);
    expect(json.errors).toHaveLength(1);
    expect(json.errors[0].rule).toBe("monotonic_dates");
  });

  it("file > 10 MB → 400 CSV_FILE_TOO_LARGE without calling validateCsv", async () => {
    const big = new File([new Uint8Array(11 * 1024 * 1024)], "x.csv");
    const req = makeMultipart(big, "daily_returns");
    const res = await POST(req);
    expect(res.status).toBe(400);
    const json = await res.json();
    expect(json.code).toBe("CSV_FILE_TOO_LARGE");
    expect(validateCsv).not.toHaveBeenCalled();
  });

  it("missing file → 400 CSV_INVALID_FORMAT", async () => {
    const req = makeMultipart(null, "daily_returns");

    expect(res.status).toBe(400);

    expect(json.code).toBe("CSV_INVALID_FORMAT");
  });

  it("bad fmt → 400 CSV_INVALID_FORMAT", async () => {
    const file = new File(["x"], "x.csv");

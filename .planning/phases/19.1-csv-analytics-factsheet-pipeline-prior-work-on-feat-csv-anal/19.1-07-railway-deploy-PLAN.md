    ```bash
    TIMEOUT_SEC=900  # 15 min — hard ceiling
    START=$(date +%s)
    while true; do
      # Replace with `mcp__railway__get_deployment_status` MCP call if available;
      # otherwise `railway status --json | jq -r '.deployments[0].status'`
      STATUS=$(mcp railway get_deployment_status --service analytics-service --pick status 2>/dev/null || echo "UNKNOWN")
      if [ "$STATUS" = "SUCCESS" ]; then break; fi
      if [ "$STATUS" = "FAILED" ] || [ "$STATUS" = "CRASHED" ]; then
        echo "FATAL: Railway deploy $STATUS" >&2
        exit 1
      fi
      if [ $(($(date +%s) - START)) -gt $TIMEOUT_SEC ]; then
        echo "FATAL: Railway deploy stuck >${TIMEOUT_SEC}s (last status=$STATUS)" >&2
        exit 1
      fi
      sleep 15
    done

    # W2: probe the deployed image SHA and assert it equals origin/main HEAD.
    # If `get_service_config --pick deployed_sha` is not the exact MCP tool, fall back to
    # `railway status --json | jq -r '.deployments[0].meta.commitHash'` (the Railway dashboard exposes the commit SHA).
    DEPLOYED_SHA=$(mcp railway get_service_config --service analytics-service --pick deployed_sha 2>/dev/null \
                   || railway status --json 2>/dev/null | jq -r '.deployments[0].meta.commitHash' \
                   || echo "UNKNOWN")
    MAIN_SHA=$(git rev-parse origin/main)
    if [ "$DEPLOYED_SHA" != "$MAIN_SHA" ]; then
      echo "FATAL: SHA mismatch deployed=$DEPLOYED_SHA main=$MAIN_SHA" >&2
      exit 1
    fi

    # Watchdog headroom invariant against the deployed source (main HEAD).
    cd analytics-service && pytest tests/test_main_worker.py::test_every_kind_has_watchdog_headroom -q || exit 1
    ```
